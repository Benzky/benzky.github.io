Module kdtree

This module is a library containing the k-d tree class.

A k-d tree, or k-dimensional binary search tree, is a data structure used for organizing points in a k-dimensional space. Because of the tree structure nearest neighbor searches only take O(log n) time.

The k-d tree in this module is an unblanaced k-d tree. This means that adding elements frequently will make the tree unbalanced and risk losing the advantages of the data structure.

Example

Example usage:

>>> from kdtree import kdtree
>>> tree = kdtree([(1,1),(-3,0.6),(100,32),(32,0x42)])
>>> tree
((32, 66), ((1, 1), ((-3, 0.6), None, None), None), ((100, 32), None, None))

>>> tree.nearest((-2,1))
(-3, 0.6)
>>> tree.nearest((50,0))
(1, 1)

>>> tree.list()
[(32, 66), (1, 1), (-3, 0.6), (100, 32)]

>>> tree.add((9,3))
>>> tree.list()
[(32, 66), (1, 1), (-3, 0.6), (9, 3), (100, 32)]

class kdtree:
    """A class for storing and searching for points in k-dimensional space.
    
    Attributes:
        point (tuple): A k-tuple representing a point in k-d space.
        left (Any): The left branch of this tree. Either another k-d tree or None.
        right (Any): The right branch of this tree. Either another k-d tree or None.
        axis (int): An integer specifying the axis this point is splitting the space.
    """
    
    def __init__(self, points, depth = 0):
        """
        Args:
            points (List[tuple]): List of k-tuples to add to the k-d tree.
            depth (int): Depth of the k-d tree.
        """
        n = len(points[0])
        
        self.axis = depth % n
        
        points.sort(key = lambda x: x[self.axis])
        mid = len(points)//2
            
        self.point = points[mid]
        self.left = kdtree(points[:mid], depth + 1) if points[:mid] else None
        self.right = kdtree(points[mid + 1:], depth + 1) if points[mid + 1:] else None
    
    def nearest(self, point, best = None):
        """Returns the point in the tree closest to the search target
        using nearest neighbor search.
        
        Args:
            point (tuple): The search target. 
            best (tuple): The point in the tree currently closest to the search target.
        
        Returns:
            tuple: The point in the tree closest to the search target.
        """
        # Bara avliva mig nu
        if self is None:
            return best
        
        if best is None:
            best = self.point
        elif self._distsquared(self.point, point) < self._distsquared(best, point):
            best = self.point
        
        best = self._bestchild(point).nearest(point, best) if self._bestchild(point) else best
        
        if self._axisdist(point) < self._distsquared(best, point):
            best = self._notbestchild(point).nearest(point,best) if self._notbestchild(point) else best
        
        return best
    
    def add(self, point):
        """Adds a point as a leaf to the correct branch of the tree.
        
        Args:
            point (tuple): The point to add.
        """
        if point[self.axis] < self.point[self.axis]:
            if self.left:
                self.left.add(point)
            else:
                self.left = kdtree([point])
        else:
            if self.right:
                self.right.add(point)
            else:
                self.right = kdtree([point])
    
    def recreate(self):
        """Rebalances the tree by recreating it from scratch.
        """
        self.__init__(self.list(), 0 if self.axis == 0 else 1)
    
    def list(self):
        """Returns an unsorted list of all the points in the tree.
        
        Returns:
            List[tuple]: An unsorted list of points in the tree.            
        """
        return self._listhelp([])
    
    def _listhelp(self, list):
        list.append(self.point)
        if self.left:
            self.left._listhelp(list)
        if self.right:
            self.right._listhelp(list)
        return list

        # alla fÃ¶rdelar med den data struktur du valt
        if point[self.axis] < self.point[self.axis]:
            return self.left if self.left else None
        return self.right if self.right else None
    
    def _axisdist(self, point):
        # Stor applÃ¥d till tuple fÃ¶r att inte kunna Ã¤ndra innehÃ¥ll
        axpoint = list(point)
        axpoint[self.axis] = self.point[self.axis]
        return self._distsquared(tuple(axpoint), point)

        if self._bestchild(point) is self.left:
            return self.right
        return self.left
    
    def _distsquared(self, a, b):
        if len(a) != len(b):
            raise "Tree contains points of different dimensions."
        return sum([(a[i]-b[i])**2 for i in range(len(a))])

        Returns:
            string: A representation of the tree.
        """
        return ", ".join([str(self.point),self.left.__repr__(),self.right.__repr__()]).join(["(",")"])
